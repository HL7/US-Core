
<!-- ================================================================
 ValueSet, CodeSystem, and OperationDefinition lists are generated by
 the included liquid script below.
 The source data is at temp/pages/_data/resources.json
 which is automatically generated by the build
 and the input/data/new_stuff which manually updated when new content is added.

It creates a flat list, sorted alphabetically by title with relative links,
and allows for highlighting new stuff using the input/data/new_stuff data file
====================================================================== -->
{% assign my_types = "" %}
{% for r_hash in site.data.resources -%}
  {% assign r_type = r_hash[0] | split: '/' | first %}
  {% assign r = r_hash[1] -%}
  {% if r_type == include.type -%}
    {% assign my_types =  my_types | append: r.name | append: "," -%}
  {% endif -%}
{% endfor %}
<ul>
{% assign my_array = my_types | split: "," | sort | uniq -%}
{% for item in my_array offset -%}
  {%- for r_hash in site.data.resources -%}
      {% assign r_type = r_hash[0] | split: '/' | first %}
      {%- assign r = r_hash[1] -%}
      {%- if r.name == item and r_type == include.type %}
        {%- assign new = false -%}
        {%- for new_stuff in site.data.new_stuff -%}
           {%- if new_stuff == item -%}
             {%- assign new = true -%}
             {%- break -%}
           {%- endif -%}
        {%- endfor -%}
        {%- if new %}
          <li><a href="{{r.path}}"><span class="bg-success" markdown="1">{{r.title}}</span><!-- new-content --></a></li>
        {% else %}
          <li><a href="{{r.path}}">{{r.title}}</a></li>
        {% endif -%}
    {% endif -%}
  {%- endfor -%}
{% endfor %}
</ul>
